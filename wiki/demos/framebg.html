jSlider.options frameBG property

Below are demonstrations of the effects of different values for jSlider's options' frameBG property.

1. rgb(170,170,170) frameBG

var	images = [
		'../../test_images/5.jpg',
		'../../test_images/6_s.jpg',
		'../../test_images/7_s.jpg',
		'../../test_images/8_s.jpg',
		'../../test_images/9_s.jpg'];

var	containerOne = $('div#jSliderOne'),
	optionsOne = {
		'frameBG': 'rgb(170,170,170)',
		'marginTop': '1em'
	},
	
	sliderOne = $.jSlider(containerOne, images, optionsOne);

2. black frameBG

var	containerTwo = $('div#jSliderTwo'),
	optionsTwo = {
		'frameBG': 'black',
		'marginTop': '1em'
	},

	sliderTwo = $.jSlider(containerTwo, images, optionsTwo);

2. #FFFFFF frameBG

var	containerThree = $('div#jSliderThree'),
	optionsThree = {
		'frameBG': '#FFFFFF',

	},

	sliderThree = $.jSlider(containerThree, images, optionsThree);
