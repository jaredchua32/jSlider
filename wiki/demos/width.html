jcSlider.options width property

Below are demonstrations of the effects of different values for jcSlider's options' width property.

1. Full Container Width

var	images = [
		'../../test_images/5.jpg',
		'../../test_images/6.jpg',
		'../../test_images/7.jpg',
		'../../test_images/8.jpg',
		'../../test_images/9.jpg'];

var	containerOne = $('div#jSliderOne'),
	optionsOne = {
		width': '100%',
		'marginTop': '1em'
	},
	
sliderOne = $.jcSlider(containerOne, images, optionsOne);

2. Fixed Container Width

var	containerTwo = $('div#jSliderTwo'),
	optionsTwo = {
		'width': '400px',
		'marginTop': '1em'
	},

sliderTwo = $.jcSlider(containerTwo, images, optionsTwo);

3. Auto Container Width

var	containerThree = $('div#jSliderThree'),
	optionsThree = {
		'width': 'auto',
		'marginTop': '1em'
	},

sliderThree = $.jcSlider(containerTwo, images, optionsThree);
